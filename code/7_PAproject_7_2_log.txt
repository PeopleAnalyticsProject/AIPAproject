import pandas as pd
from google import genai
from google.genai import types
import time

# 1. API 키 및 클라이언트 설정
API_KEY = "YOUR_API_KEY_HERE" 
client = genai.Client(api_key=API_KEY)

# 2. 엑셀 파일 로드
file_path = r'C:\Users\user\Desktop\7_Paproject_7_2_log.xlsx'
df = pd.read_excel(file_path)

def classify_log(text):
    if pd.isna(text) or str(text).strip() == "":
        return "N/A"
    
    prompt = f"""
    당신은 사내 메신저 대화 내용을 분석하는 전문가입니다.
    다음 문장을 'advice', 'communication', 'collaboration' 중 하나로 분류하세요.
    
    - advice: 조언, 상담, 가이드 제공
    - communication: 일반적인 대화, 정보 공유, 안부
    - collaboration: 공동 작업, 프로젝트 협업, 업무 요청
    
    분류 결과 외에 다른 설명은 절대 하지 말고 단어 하나만 출력하세요.
    
    문장: {text}
    """
    
    try:
        # 사용자가 지정하신 gemini-2.5-flash 모델 사용
        response = client.models.generate_content(
            model="gemini-2.5-flash",
            contents=prompt
        )
        
        result = response.text.strip().lower()
        return result
    
    except Exception as e:
        print(f"에러 발생: {e}")
        return "error"

# 3. 데이터 프레임의 'log' 컬럼(소문자)에 적용
print("분류 작업을 시작합니다...")

# 컬럼명 'log'로 수정 완료
df['type'] = df['log'].apply(classify_log)

# 4. 결과 저장
output_path = r'C:\Users\user\Desktop\7_Paproject_7_2_log_classified.xlsx'
df.to_excel(output_path, index=False)

print(f"완료! 파일 저장 경로: {output_path}")
print(df[['log', 'type']].head()) # 결과 확인 시에도 소문자 log 사용
