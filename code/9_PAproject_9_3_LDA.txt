import pandas as pd
import re
from soynlp.noun import LRNounExtractor_v2
from sklearn.feature_extraction.text import CountVectorizer
from sklearn.decomposition import LatentDirichletAllocation

# 1. 파일 경로 설정
csv_path = r'C:\Users\user\Desktop\9_PAproject_9_3_review.csv'
stopwords_path = r'C:\Users\user\Desktop\9_stopwords.txt'

# 2. 데이터 및 불용어 로드
df = pd.read_csv(csv_path)
reviews = df['review'].astype(str).tolist()

with open(stopwords_path, 'r', encoding='utf-8') as f:
    stopwords = [line.strip() for line in f if line.strip()]

# 3. soynlp 명사 추출 및 전처리
noun_extractor = LRNounExtractor_v2(verbose=False)
nouns_dict = noun_extractor.train_extract(reviews)
learned_nouns = set(nouns_dict.keys())

def get_nouns(text):
    text = re.sub(r"[^가-힣 ]", " ", text)
    words = text.split()
    return [w for w in words if w in learned_nouns and w not in stopwords and len(w) > 1]

tokenized_reviews = [" ".join(get_nouns(doc)) for doc in reviews]
valid_indices = [i for i, doc in enumerate(tokenized_reviews) if doc.strip()]
final_reviews = [tokenized_reviews[i] for i in valid_indices]
original_df_subset = df.iloc[valid_indices].copy()

# 4. LDA 모델링 (k=5)
vectorizer = CountVectorizer(max_df=0.95, min_df=2)
X = vectorizer.fit_transform(final_reviews)

lda = LatentDirichletAllocation(n_components=5, max_iter=20, random_state=42)
lda_output = lda.fit_transform(X)

# 5. 결과물 1: 토픽별 핵심 단어 정리
vocab = vectorizer.get_feature_names_out()
topic_keywords = []
for i, topic in enumerate(lda.components_):
    top_words = [vocab[idx] for idx in topic.argsort()[:-11:-1]]
    topic_keywords.append(top_words)
    print(f"Topic #{i+1}: {' | '.join(top_words)}")

# 6. 결과물 2: 각 리뷰가 어떤 토픽에 속하는지 정리 (CSV 저장)
doc_topic_df = pd.DataFrame(lda_output, columns=[f'Topic_{i+1}' for i in range(5)])
result_df = pd.concat([original_df_subset.reset_index(drop=True), doc_topic_df], axis=1)

# 가장 비중이 높은 토픽 번호 부여
result_df['Dominant_Topic'] = result_df[[f'Topic_{i+1}' for i in range(5)]].idxmax(axis=1)

# 결과 저장
result_df.to_csv(r'C:\Users\user\Desktop\LDA_Result_Analysis.csv', index=False, encoding='utf-8-sig')
print("\n" + "="*50)
print("✅ 분석 결과가 'LDA_Result_Analysis.csv'로 저장되었습니다.")
