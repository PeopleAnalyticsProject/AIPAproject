import requests
from bs4 import BeautifulSoup
import pandas as pd
from google import genai
import json

# --- 설정 ---
NAVER_CLIENT_ID = "YOUR_NAVER_CLIENT_ID"
NAVER_CLIENT_SECRET = "YOUR_NAVER_CLIENT_SECRET"
GEMINI_API_KEY = "YOUR_GEMINI_API_KEY"

client = genai.Client(api_key=GEMINI_API_KEY)
MODEL_ID = "gemini-2.5-flash"

def get_naver_news(query, display=20):
    url = f"https://openapi.naver.com/v1/search/news.json?query={query}&display={display}"
    headers = {
        "X-Naver-Client-Id": NAVER_CLIENT_ID,
        "X-Naver-Client-Secret": NAVER_CLIENT_SECRET
    }
    response = requests.get(url, headers=headers)
    return response.json().get('items', [])

# 1. 뉴스 데이터 수집 및 텍스트 정제
search_query = "적재적소 인력배치 사례"
news_items = get_naver_news(search_query, display=20)

news_list_text = ""
for i, item in enumerate(news_items):
    title = BeautifulSoup(item['title'], "html.parser").get_text()
    desc = BeautifulSoup(item['description'], "html.parser").get_text()
    news_list_text += f"[{i+1}] 제목: {title}\n내용: {desc}\n\n"

# 2. Gemini에게 한꺼번에 분석 요청 (프롬프트 최적화)
prompt = f"""
다음은 뉴스 목록입니다. 이 중에서 '적재적소의 인력배치'와 관련된 구체적인 사례들을 찾아 정리해주세요.
결과는 반드시 아래의 JSON 리스트 형식으로만 출력해주세요. 관련 사례가 없다면 빈 리스트 []를 출력하세요.

JSON 형식:
[
  {{"기업_기관명": "이름", "배치_내용": "내용 요약", "기대_효과": "효과 요약", "핵심_키워드": "키워드"}}
]

뉴스 목록:
{news_list_text}
"""

print("Gemini가 전체 리스트를 통합 분석 중입니다...")

try:
    # 텍스트가 길어도 Flash 모델은 한 번에 처리가 가능합니다.
    response = client.models.generate_content(
        model=MODEL_ID,
        contents=prompt,
        config={'response_mime_type': 'application/json'} # JSON 응답 강제
    )
    
    # 3. 결과 파싱 및 엑셀 저장
    raw_data = json.loads(response.text)
    
    if raw_data:
        df = pd.DataFrame(raw_data)
        file_name = "인력배치_사례_통합분석.xlsx"
        df.to_excel(file_name, index=False)
        print(f"분석 완료! {len(df)}개의 사례가 '{file_name}'에 저장되었습니다.")
        print(df.head()) # 결과 상단 노출
    else:
        print("관련된 인력배치 사례를 찾지 못했습니다.")

except Exception as e:
    print(f"오류가 발생했습니다: {e}")
