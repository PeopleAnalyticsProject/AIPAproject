import pandas as pd
from soynlp.noun import LRNounExtractor_v2
from sklearn.feature_extraction.text import TfidfVectorizer

# 1. 파일 경로 설정
csv_path = r'C:\Users\user\Desktop\9_Paproject_9_3_review.csv'
stopwords_path = r'C:\Users\user\Desktop\9_stopwords.txt'

# 2. 데이터 및 불용어 불러오기
df = pd.read_csv(csv_path)
reviews = df['review'].astype(str).tolist()

with open(stopwords_path, 'r', encoding='utf-8') as f:
    stopwords = [line.strip() for line in f if line.strip()]

# 3. soynlp 명사 추출 및 학습
noun_extractor = LRNounExtractor_v2(verbose=False)
# 전체 코퍼스에서 명사 후보와 점수를 미리 계산합니다.
nouns_dict = noun_extractor.train_extract(reviews)

# 학습된 명사 리스트만 따로 저장 (추출된 단어들)
learned_nouns = set(nouns_dict.keys())


# 4. 전처리 함수 수정 (에러 발생 지점 해결)
def get_only_learned_nouns(text):
    # 문장을 띄어쓰기 단위로 자른 후, 학습된 명사 사전에 있는 단어만 필터링
    words = text.split()
    return [w for w in words if w in learned_nouns and w not in stopwords and len(w) > 1]


# 전체 문서에 대해 명사 리스트 생성
noun_documents = [" ".join(get_only_learned_nouns(doc)) for doc in reviews]

# 5. Scikit-learn을 이용한 TF-IDF 계산
vectorizer = TfidfVectorizer()
# 간혹 모든 단어가 필터링되어 빈 문서가 생길 경우를 대비해 예외 처리
try:
    tfidf_matrix = vectorizer.fit_transform(noun_documents)

    # 6. TF-IDF 기준 상위 20개 단어 추출
    tfidf_scores = tfidf_matrix.mean(axis=0).tolist()[0]
    features = vectorizer.get_feature_names_out()

    word_score_list = [(word, score) for word, score in zip(features, tfidf_scores)]
    top_20_words = sorted(word_score_list, key=lambda x: x[1], reverse=True)[:20]

    # 결과 출력
    print(f"{'Rank':<5} | {'Word':<15} | {'TF-IDF Score':<10}")
    print("-" * 40)
    for i, (word, score) in enumerate(top_20_words, 1):
        print(f"{i:<5} | {word:<15} | {score:.4f}")

except ValueError:
    print("TF-IDF를 계산할 수 있는 단어가 부족합니다. 데이터나 불용어 리스트를 확인해 주세요.")
