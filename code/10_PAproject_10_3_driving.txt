import pandas as pd
import requests
import datetime

# 1. 파일 경로 및 회사 주소 설정
file_path = r"C:\Users\user\Desktop\10_PAproject_10_3_Address.xlsx"
output_path = r"C:\Users\user\Desktop\10_PAproject_10_3_DriveTime_Result.xlsx"
COMPANY_ADDRESS = "서울특별시 종로구 청계천로 1"

# 2. 네이버 API 인증 정보 입력
CLIENT_ID = "YOUR_NAVER_CLIENT_ID"       # 여기에 발급받은 Client ID 입력
CLIENT_SECRET = "YOUR_NAVER_CLIENT_SECRET" # 여기에 발급받은 Client Secret 입력

headers = {
    "X-NCP-APIGW-API-KEY-ID": CLIENT_ID,
    "X-NCP-APIGW-API-KEY": CLIENT_SECRET
}

# 3. 출발/퇴근 시간 설정 (ISO 8601 형식: YYYY-MM-DDTHH:MM:SS)
# 미래의 특정 시점 교통 예측을 위해 다가오는 평일(또는 오늘) 날짜를 기준으로 계산합니다.
today = datetime.date.today()
# 주말(토:5, 일:6)일 경우 다음주 월요일로 설정
if today.weekday() == 5:
    target_date = today + datetime.timedelta(days=2)
elif today.weekday() == 6:
    target_date = today + datetime.timedelta(days=1)
else:
    target_date = today

departure_time_am = f"{target_date}T07:30:00"  # 출근 출발시간 (07:30)
departure_time_pm = f"{target_date}T18:30:00"  # 퇴근 출발시간 (18:30)


# 4. 네이버 지오코딩 API: 주소 -> 위경도(좌표) 변환 함수
def get_coordinates(address):
    url = "https://maps.apigw.ntruss.com/map-geocode/v2/geocode"
    params = {"query": address}
    try:
        response = requests.get(url, headers=headers, params=params)
        if response.status_code == 200:
            res_json = response.json()
            if res_json.get("addresses"):
                # Driving API는 "경도,위도" 순서의 문자열 포맷을 주로 사용하므로 콤마로 결합해 리턴
                lon = res_json["addresses"][0]["x"]
                lat = res_json["addresses"][0]["y"]
                return f"{lon},{lat}"
        else:
            print(f"[지오코딩 실패] 상태코드 {response.status_code} - 주소: {address}")
    except Exception as e:
        print(f"[지오코딩 에러] {e} - 주소: {address}")
    return None


# 5. 네이버 Driving API: 두 좌표 간 이동 시간 계산 함수 (반환값: 분)
def get_drive_time(start_coord, end_coord, departure_time):
    url = "https://maps.apigw.ntruss.com/map-direction/v1/driving"
    params = {
        "start": start_coord,
        "goal": end_coord,
        "departure-time": departure_time
    }
    try:
        response = requests.get(url, headers=headers, params=params)
        if response.status_code == 200:
            res_json = response.json()
            # traoptimal(실시간 최적 경로) 결과 추출
            if res_json.get("route") and res_json["route"].get("traoptimal"):
                # 기본 반환 단위는 밀리초(ms)이므로, 분(minute) 단위로 변환 (반올림)
                duration_ms = res_json["route"]["traoptimal"][0]["summary"]["duration"]
                duration_min = round(duration_ms / 1000 / 60, 1)
                return duration_min
        else:
            print(f"[경로조회 실패] 상태코드 {response.status_code}")
    except Exception as e:
        print(f"[경로조회 에러] {e}")
    return None


# ==========================================
# 메인 로직 시작
# ==========================================

# 1) 데이터 불러오기 및 gtw가 'driving'인 직원만 필터링
print("1. 엑셀 파일을 읽어오는 중...")
df = pd.read_excel(file_path)
df_driving = df[df['gtw'] == 'driving'].copy()

if df_driving.empty:
    print("[알림] gtw 변수가 'driving'인 직원이 없습니다. 코드를 종료합니다.")
else:
    print(f"총 {len(df_driving)}명의 driving 대상 직원을 찾았습니다.")
    
    # 2) 회사 위치 좌표 변환
    print(f"2. 회사 위치 좌표 변환 중 ({COMPANY_ADDRESS})...")
    company_coord = get_coordinates(COMPANY_ADDRESS)
    
    if not company_coord:
        print("[오류] 회사 주소의 좌표 변환에 실패했습니다. API 키나 주소를 확인하세요.")
    else:
        am_times = []
        pm_times = []
        total_times = []
        
        # 3) 직원 집 주소별 출퇴근 시간 계산
        print("3. 직원별 출퇴근 운전 시간 계산 중 (API 호출)...")
        for idx, row in df_driving.iterrows():
            home_address = row['address']
            home_coord = get_coordinates(home_address)
            
            if home_coord:
                # 출근: 집(home) -> 회사(company), 아침 7:30
                am_time = get_drive_time(home_coord, company_coord, departure_time_am)
                # 퇴근: 회사(company) -> 집(home), 저녁 18:30
                pm_time = get_drive_time(company_coord, home_coord, departure_time_pm)
                
                if am_time is not None and pm_time is not None:
                    am_times.append(am_time)
                    pm_times.append(pm_time)
                    total_times.append(round(am_time + pm_time, 1))
                else:
                    am_times.append(None)
                    pm_times.append(None)
                    total_times.append(None)
            else:
                print(f"[경고] 직원 주소 변환 실패 - {home_address}")
                am_times.append(None)
                pm_times.append(None)
                total_times.append(None)
                
        # 4) 계산 결과를 데이터프레임에 추가
        df_driving['출근시간_분'] = am_times
        df_driving['퇴근시간_분'] = pm_times
        df_driving['총출퇴근시간_분'] = total_times
        
        # 5) 결과를 별도의 엑셀 파일로 저장
        df_driving.to_excel(output_path, index=False)
        print(f"\n4. 계산 완료! 결과가 다음 경로에 저장되었습니다:\n{output_path}")
