import pandas as pd
from soynlp.tokenizer import MaxScoreTokenizer
from soynlp.noun import LRNounExtractor_v2
from collections import Counter

# 1. 파일 경로 설정
csv_path = r'C:\Users\user\Desktop\9_PAproject_9_3_review.csv'
stopwords_path = r'C:\Users\user\Desktop\9_stopwords.txt'

# 2. 데이터 불러오기
# 인코딩 에러 발생 시 encoding='cp949' 또는 'utf-8-sig'로 변경해 보세요.
df = pd.read_csv(csv_path)
reviews = df['review'].astype(str).tolist()

# 3. 불용어 리스트 로드
with open(stopwords_path, 'r', encoding='utf-8') as f:
    stopwords = [line.strip() for line in f if line.strip()]

# 4. soynlp를 이용한 단어 추출 및 토큰화
# 명사 추출기를 통해 학습데이터 구축 (soynlp의 특징)
noun_extractor = LRNounExtractor_v2(verbose=True)
nouns = noun_extractor.train_extract(reviews)

# 추출된 명사 점수를 바탕으로 토크나이저 생성
scores = {word: score.score for word, score in nouns.items()}
tokenizer = MaxScoreTokenizer(scores=scores)

# 5. 전처리 및 토큰화 (불용어 제거 포함)
tokenized_reviews = []
for doc in reviews:
    tokens = tokenizer.tokenize(doc)
    # 불용어 제거 및 단일 글자 제거 (선택 사항)
    clean_tokens = [w for w in tokens if w not in stopwords and len(w) > 1]
    tokenized_reviews.append(clean_tokens)

# 6. 2-gram (Bigram) 생성
bigrams = []
for tokens in tokenized_reviews:
    # 한 문장에서 연속된 두 단어를 조합
    grams = [f"{tokens[i]} {tokens[i+1]}" for i in range(len(tokens)-1)]
    bigrams.extend(grams)

# 7. 빈도 분석 결과 출력
bigram_counts = Counter(bigrams)
print(f"{'Bigram':<20} | {'Frequency':<10}")
print("-" * 35)
for gram, count in bigram_counts.most_common(20):
    print(f"{gram:<20} | {count:<10}")
